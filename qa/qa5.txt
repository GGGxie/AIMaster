```python
import pandas as pd

# 读取Excel文件
df = pd.read_excel('./data/2023年8月-9月销售记录.xlsx')

# 筛选2023年8月和9月的数据
df['销售日期'] = pd.to_datetime(df['销售日期'])  # 确保销售日期列是datetime类型
august_data = df[(df['销售日期'] >= '2023-08-01') & (df['销售日期'] <= '2023-08-31')]
september_data = df[(df['销售日期'] >= '2023-09-01') & (df['销售日期'] <= '2023-09-30')]

# 计算每月的总销售量和总销售额
august_total_sales_volume = august_data['销售量'].sum()
august_total_sales_amount = (august_data['销售量'] * august_data['单价(元)']).sum()

september_total_sales_volume = september_data['销售量'].sum()
september_total_sales_amount = (september_data['销售量'] * september_data['单价(元)']).sum()

# 输出结果
print(f"2023年8月的总销售量为：{august_total_sales_volume}，总销售额为：{august_total_sales_amount}元")
print(f"2023年9月的总销售量为：{september_total_sales_volume}，总销售额为：{september_total_sales_amount}元")
```
